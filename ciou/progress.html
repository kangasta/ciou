 22        if not config:

 33    def _run(self):
 34        while True:
 35            item = self._queue.get()
 36            if isinstance(item, Update):
 37                self._store.push(item)
 38            if item == STOP:
 39                self._store.close()
 40                self._renderer.render(self._store)
 41                break
 42            else:
 43                self._renderer.render(self._store)
 44
 45    def start(self):
 46        '''Start the rendering loop on a new thread.
 47        '''
 48        self._thread = Thread(
 49            target=self._run
 50        )
 51        self._thread.start()
 52
 53        self._ticker = Ticker(SCREEN_WRITE_INTERVAL, self._queue)
 54        self._ticker.start()
 55
 56    def push(self, update: Update):
 57        '''Push `Update` to the progress log.
 58        '''
 59        self._queue.put(update)
 60
 61    def stop(self):
 62        '''Stop the rendering loop after rendering the final state of the
 63        progress log.
 64        '''
 65        self._ticker.stop()
 66
 67        self._queue.put(STOP)
 68
 69        if self._thread:
 70            self._thread.join()
 71            self._thread = None
